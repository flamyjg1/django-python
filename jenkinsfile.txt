pipeline {
    agent any

    stages {
        stage('GIT') {
            steps {
                echo 'To clone git repisotory'
                checkout scmGit(branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/flamyjg1/django-project']])
            }
        }
        stage('Dependencies') {
            steps {
                echo 'installing dependencies for django project'
                bat 'pip install -r requirements.txt'
            }
        }
         stage('Build') {
            steps {
                echo 'check for application database'
                bat 'python manage.py makemigrations'
                
                echo 'Building the application database'
                bat 'python manage.py migrate'
            }
        }
         stage('Test') {
            steps {
                echo 'Run test cases'
                bat 'python manage.py test'
              
            }
        }
    }
}
